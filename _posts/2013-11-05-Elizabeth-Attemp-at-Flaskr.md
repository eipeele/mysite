In my flaskr.py

```python

@app.route('/logout')
def logout():
    session.pop('logged_in', None)
    flash('You were logged out')
    return redirect(url_for('show_entries'))

if __name__ == '__main__':
    init_db()
    app.run()

@app.route('/about')
def about():
    return render_template('about.html')

```
